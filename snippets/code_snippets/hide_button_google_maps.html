hide button google maps

Button zum verstecken von Google Maps.

Anpassen:

  • gegebenenfalls CSS
  • API-Key in .loadScript-Befehl
  • MapPin sowie Daten für Marker

Javascript

$(document).ready(function(){

	//JQuery load script function - loads maps-script and calls showMap() when script is loaded
	jQuery.loadScript = function (url, callback) {
		jQuery.ajax({
			url: url,
			dataType: 'script',
			success: callback,
			async: true
		});
	}

	if (getCookie('gmaps') == 'yes') {
		if (typeof someObject == 'undefined') $.loadScript('https://maps.google.com/maps/api/js?key=AIzaSyAtL0gCWVVrTCddSSy4A0IdHukOS_RTG9Y', function(){
			showMap();
		})
	}
	$('#show-map').click(function() {
		document.cookie = "gmaps = yes";
		if (typeof someObject == 'undefined') $.loadScript('https://maps.google.com/maps/api/js?key=AIzaSyAtL0gCWVVrTCddSSy4A0IdHukOS_RTG9Y', function(){
			showMap();
		})
	});




	//showMap() shows the map, if maps script is loaded
	function showMap() {
		var latlng = new google.maps.LatLng(50.1877076, 8.587860399999954);

		var myOptions = {
			zoom: 17,
			center: latlng,
			mapTypeId: google.maps.MapTypeId.ROADMAP,
			styles: _app.mapStyle
		};


		var map = new google.maps.Map(document.getElementById("map_canvas"), myOptions);

		var image = '../bilder/navigat/kartenpin.png';
		var marker = new google.maps.Marker({
			title:"VOGT & SOMMER GmbH",
			icon: image,
			map: map,
			place: {
				location: latlng,
				placeId: 'ChIJJ5hAw6HhvkcRaj5gn0QGLjM'
			}

		});


		var contentString = '<div id="Tooltip">'+
			'<h1>VOGT & SOMMER GmbH</h1>'+
			'<p>In den Schwarzwiesen 23<br>'+
			'61440 Oberursel</p>'+
			'</div>';

		var infowindow = new google.maps.InfoWindow({
			content: contentString
		});

		marker.addListener('click', function() {
			infowindow.open(map, marker);
		});
	}


	//sets cookie, so maps is loaded everytime the page is visited. Cookie deletes if browser is closed.
	function getCookie(cname) {
		let name = cname + "=";
		let decodedCookie = decodeURIComponent(document.cookie);
		let ca = decodedCookie.split(';');
		for(let i = 0; i <ca.length; i++) {
			let c = ca[i];
			while (c.charAt(0) == ' ') {
				c = c.substring(1);
			}
			if (c.indexOf(name) == 0) {
				return c.substring(name.length, c.length);
			}
		}
		return "";
	}

});
